    OptionsQuery

    $obj = new MslsGetSet;
$obj->tmp = 'test';
$val = $obj->get_arr();          // array( 'tmp' => 'test' ) == $val
$val = $obj->has_value( 'tmp' ); // true == $val
$val = $obj->is_empty();         // false == $val
echo $obj->tmp;                  // echos 'test'
$obj->reset();
$val = $obj->get_arr();          // array() == $val
$val = $obj->has_value( 'tmp' ); // false == $val
$val = $obj->is_empty();         // true == $val
echo $obj->tmp;                  // echos null

    has_value(string $key) : bool
    Inherited

    This is method is similar to the overloaded __isset-method since __set cleans empty properties but I use for example $obj->has_value( $temp ) and not isset( $obj->$temp ) which is the same but a little bit ugly.
