  8: /**
  9:  * Generic class for overloading properties
 10:  *
 11:  * <code>
 12:  * $obj = new MslsGetSet;
 13:  * $obj->tmp = 'test';
 14:  * $val = $obj->get_arr();          // array( 'tmp' => 'test' ) == $val
 15:  * $val = $obj->has_value( 'tmp' ); // true == $val
 16:  * $val = $obj->is_empty();         // false == $val
 17:  * echo $obj->tmp;                  // prints 'test'
 18:  * $obj->reset();
 19:  * $val = $obj->get_arr();          // array() == $val
 20:  * $val = $obj->has_value( 'tmp' ); // false == $val
 21:  * $val = $obj->is_empty();         // true == $val
 22:  * echo $obj->tmp;                  // prints null
 23:  * </code>
 24:  * @package Msls
 25:  */
 26: class MslsGetSet {
 27: 
 28:     /**
 29:      * A generic container for all properties of an instance
 30:      * @var array $arr
 31:      */
 32:     protected $arr = array();
 33: 
 34:     /**
 35:      * Overloaded set-method 
 36:      * @param mixed $key
 37:      * @param mixed $value
 38:      */
 39:     final public function __set( $key, $value ) {
 40:         $this->arr[$key] = $value;
 41:         if ( empty( $this->arr[$key] ) )
 42:             unset( $this->arr[$key] );
 43:     }
 44: 
 45:     /**
 46:      * Overloaded get-method
 47:      * @param mixed $key
 48:      * @return mixed
 49:      */
 50:     final public function __get( $key ) {
 51:         return( isset( $this->arr[$key] ) ? $this->arr[$key] : null );
 52:     }
 53: 
 54:     /**
 55:      * Overloaded isset-method
 56:      * @param mixed $key
 57:      * @return bool
 58:      */
 59:     final public function __isset( $key ) {
 60:         return isset( $this->arr[$key] );
 61:     }
 62: 
 63:     /**
 64:      * Overloaded unset-method
 65:      * @param mixed $key
 66:      */
 67:     final public function __unset( $key ) {
 68:         if ( isset( $this->arr[$key] ) )
 69:             unset( $this->arr[$key] );
 70:     }
 71: 
 72:     /**
 73:      * Reset the properties-container to an empty array
 74:      * @return MslsGetSet
 75:      */
 76:     public function reset() {
 77:         $this->arr = array();
 78:         return $this;
 79:     }
 80: 
 81:     /**
 82:      * Check if the array has an non empty item with the specified key
 83:      * 
 84:      * This is method is similar to the overloaded __isset-method since
 85:      * __set cleans empty properties but I use for example 
 86:      * $obj->has_value( $temp ) and not isset( $obj->$temp ) which is
 87:      * the same but a little bit ugly.
 88:      * @param string $key
 89:      * @return bool
 90:      */ 
 91:     public function has_value( $key ) {
 92:         return( !empty( $this->arr[$key] ) );
 93:     }
 94: 
 95:     /**
 96:      * Check if the properties-container is empty
 97:      * @return bool
 98:      */ 
 99:     public function is_empty() {
100:         return( empty( $this->arr ) );
101:     }
102: 
103:     /**
104:      * Get the complete properties-container as an array
105:      * @return array
106:      */
107:     final public function get_arr() {
108:         return $this->arr;
109:     }
110: 
111: }
