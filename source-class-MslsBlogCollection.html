 51:         $options                   = MslsOptions::instance();

 53:         $this->current_blog_output = isset( $options->output_current_blog );
 54:         $this->objects_order       = $options->get_order();
 55:         if ( !$options->is_excluded() ) {
 56:             if ( has_filter( 'msls_blog_collection_construct' ) ) {
 57:                 $blogs_collection = apply_filters(
 58:                     'msls_blog_collection_construct',
 59:                     array()
 60:                 );
 61:             }
 62:             else {
 63:                 $args = array(
 64:                     'blog_id' => $this->current_blog_id,
 65:                     'orderby' => 'registered',
 66:                     'fields' => 'ID',
 67:                 );
 68:                 $blogs_collection = array();
 69:                 $blog_users       = get_users( $args );
 70:                 if ( !empty ( $blog_users ) )
 71:                     $blogs_collection = get_blogs_of_user( $blog_users[0] );
 72:             }
 73:             foreach ( (array) $blogs_collection as $blog ) {
 74:                 /*
 75:                  * get_user_id_from_string returns objects with userblog_id-members 
 76:                  * instead of a blog_id ... so we need just some correction ;)
 77:                  *
 78:                  */
 79:                 if ( !isset( $blog->userblog_id ) && isset( $blog->blog_id) )
 80:                     $blog->userblog_id = $blog->blog_id;
 81:                 if ( $blog->userblog_id != $this->current_blog_id ) {
 82:                     $temp = get_blog_option( $blog->userblog_id, 'msls' );
 83:                     if ( is_array( $temp ) && empty( $temp['exclude_current_blog'] ) && $this->is_plugin_active( $blog->userblog_id ) )
 84:                         $this->objects[$blog->userblog_id] = new MslsBlog(
 85:                             $blog,
 86:                             $temp['description']
 87:                         );
 88:                 }
 89:                 else {
 90:                     $this->objects[$this->current_blog_id] = new MslsBlog(
 91:                         $blog,
 92:                         $options->description
 93:                     );
 94:                 }
 95:             }
 96:             uasort( $this->objects, array( 'MslsBlog', $this->objects_order ) );
 97:         }
 98:     }

143:         if ( !is_array( $this->active_plugins ) )
